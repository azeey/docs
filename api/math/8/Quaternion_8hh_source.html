170 public: inline Quaternion<T> Inverse() const
171 {
172 T s = 0;
173 Quaternion<T> q(this->qw, this->qx, this->qy, this->qz);
174
175 // use s to test if quaternion is valid
176 s = q.qw * q.qw + q.qx * q.qx + q.qy * q.qy + q.qz * q.qz;
177
178 if (equal<T>(s, static_cast<T>(0)))
179 {
180 q.qw = 1.0;
181 q.qx = 0.0;
182 q.qy = 0.0;
183 q.qz = 0.0;
184 }
185 else
186 {
187 // deal with non-normalized quaternion
188 // div by s so q * qinv = identity
189 q.qw = q.qw / s;
190 q.qx = -q.qx / s;
191 q.qy = -q.qy / s;
192 q.qz = -q.qz / s;
193 }
194 return q;
195 }

204 public: Quaternion<T> Log() const
205 {
206 Quaternion<T> result;
207 result.qw = 0.0;
208
209 if (std::abs(this->qw) < 1.0)
210 {
211 T fAngle = acos(this->qw);
212 T fSin = sin(fAngle);
213 if (std::abs(fSin) >= 1e-3)
214 {
215 T fCoeff = fAngle/fSin;
216 result.qx = fCoeff*this->qx;
217 result.qy = fCoeff*this->qy;
218 result.qz = fCoeff*this->qz;
219 return result;
220 }
221 }
222
223 result.qx = this->qx;
224 result.qy = this->qy;
225 result.qz = this->qz;
226
227 return result;

362 public: void SetFromEuler(T _roll, T _pitch, T _yaw)
363 {
364 T phi, the, psi;
365
366 phi = _roll / T(2.0);
367 the = _pitch / T(2.0);
368 psi = _yaw / T(2.0);
369
370 this->qw = T(cos(phi) * cos(the) * cos(psi) +
371 sin(phi) * sin(the) * sin(psi));
372 this->qx = T(sin(phi) * cos(the) * cos(psi) -
373 cos(phi) * sin(the) * sin(psi));
374 this->qy = T(cos(phi) * sin(the) * cos(psi) +
375 sin(phi) * cos(the) * sin(psi));
376 this->qz = T(cos(phi) * cos(the) * sin(psi) -
377 sin(phi) * sin(the) * cos(psi));
378
379 this->Normalize();
380 }

384 public: Vector3<T> Euler() const
385 {
386 Vector3<T> vec;
387
388 T tol = static_cast<T>(1e-15);
389
390 Quaternion<T> copy = *this;
391 T squ;
392 T sqx;
393 T sqy;
394 T sqz;
395
396 copy.Normalize();
397
398 squ = copy.qw * copy.qw;
399 sqx = copy.qx * copy.qx;
400 sqy = copy.qy * copy.qy;
401 sqz = copy.qz * copy.qz;
402
403 // Pitch
404 T sarg = -2 * (copy.qx*copy.qz - copy.qw * copy.qy);
405 if (sarg <= T(-1.0))
406 {
407 vec.Y(T(-0.5*GZ_PI));
408 }
409 else if (sarg >= T(1.0))
410 {
411 vec.Y(T(0.5*GZ_PI));
412 }
413 else
414 {
415 vec.Y(T(asin(sarg)));
416 }
417
418 // If the pitch angle is PI/2 or -PI/2, we can only compute
419 // the sum roll + yaw. However, any combination that gives
420 // the right sum will produce the correct orientation, so we
421 // set yaw = 0 and compute roll.
422 // pitch angle is PI/2
423 if (std::abs(sarg - 1) < tol)
424 {
425 vec.Z(0);
426 vec.X(T(atan2(2 * (copy.qx*copy.qy - copy.qz*copy.qw),
427 squ - sqx + sqy - sqz)));
428 }
429 // pitch angle is -PI/2
430 else if (std::abs(sarg + 1) < tol)
431 {
432 vec.Z(0);
433 vec.X(T(atan2(-2 * (copy.qx*copy.qy - copy.qz*copy.qw),
434 squ - sqx + sqy - sqz)));
435 }
436 else
437 {
438 // Roll
439 vec.X(T(atan2(2 * (copy.qy*copy.qz + copy.qw*copy.qx),
440 squ - sqx - sqy + sqz)));
441
442 // Yaw
443 vec.Z(T(atan2(2 * (copy.qx*copy.qy + copy.qw*copy.qz),
444 squ + sqx - sqy - sqz)));
445 }
446
447 return vec;
448 }

517 public: void SetFromMatrix(const Matrix3<T> &_mat)
518 {
519 const T trace = _mat(0, 0) + _mat(1, 1) + _mat(2, 2);
520 if (trace > 0.0000001)
521 {
522 qw = sqrt(1 + trace) / 2;
523 const T s = 1.0 / (4 * qw);
524 qx = (_mat(2, 1) - _mat(1, 2)) * s;
525 qy = (_mat(0, 2) - _mat(2, 0)) * s;
526 qz = (_mat(1, 0) - _mat(0, 1)) * s;
527 }
528 else if (_mat(0, 0) > _mat(1, 1) && _mat(0, 0) > _mat(2, 2))
529 {
530 qx = sqrt(1.0 + _mat(0, 0) - _mat(1, 1) - _mat(2, 2)) / 2;
531 const T s = 1.0 / (4 * qx);
532 qw = (_mat(2, 1) - _mat(1, 2)) * s;
533 qy = (_mat(1, 0) + _mat(0, 1)) * s;
534 qz = (_mat(0, 2) + _mat(2, 0)) * s;
535 }
536 else if (_mat(1, 1) > _mat(2, 2))
537 {
538 qy = sqrt(1.0 - _mat(0, 0) + _mat(1, 1) - _mat(2, 2)) / 2;
539 const T s = 1.0 / (4 * qy);
540 qw = (_mat(0, 2) - _mat(2, 0)) * s;
541 qx = (_mat(0, 1) + _mat(1, 0)) * s;
542 qz = (_mat(1, 2) + _mat(2, 1)) * s;
543 }
544 else
545 {
546 qz = sqrt(1.0 - _mat(0, 0) - _mat(1, 1) + _mat(2, 2)) / 2;
547 const T s = 1.0 / (4 * qz);
548 qw = (_mat(1, 0) - _mat(0, 1)) * s;
549 qx = (_mat(0, 2) + _mat(2, 0)) * s;
550 qy = (_mat(1, 2) + _mat(2, 1)) * s;
551 }

563 public: void SetFrom2Axes(const Vector3<T> &_v1,
564 const Vector3<T> &_v2)
565 {
566 // generally, we utilize the fact that a quat (w, x, y, z) represents
567 // rotation of angle 2*w about axis (x, y, z)
568 //
569 // so we want to take get a vector half-way between no rotation and the
570 // double rotation, which is
571 // [ (1, (0, 0, 0)) + (_v1 dot _v2, _v1 x _v2) ] / 2
572 // if _v1 and _v2 are unit quaternions
573 //
574 // since we normalize the result anyway, we can omit the division,
575 // getting the result:
576 // [ (1, (0, 0, 0)) + (_v1 dot _v2, _v1 x _v2) ].Normalized()
577 //
578 // if _v1 and _v2 are not normalized, the magnitude (1 + _v1 dot _v2)
579 // is multiplied by k = norm(_v1)*norm(_v2)
580
581 const T kCosTheta = _v1.Dot(_v2);
582 const T k = sqrt(_v1.SquaredLength() * _v2.SquaredLength());
583
584 if (fabs(kCosTheta/k + 1) < 1e-6)
585 {
586 // the vectors are opposite
587 // any vector orthogonal to _v1
588 Vector3<T> other;
589 {
590 const Vector3<T> _v1Abs(_v1.Abs());
591 if (_v1Abs.X() < _v1Abs.Y())
592 {
593 if (_v1Abs.X() < _v1Abs.Z())
594 {
595 other.Set(1, 0, 0);
596 }
597 else
598 {
599 other.Set(0, 0, 1);
600 }
601 }
602 else
603 {
604 if (_v1Abs.Y() < _v1Abs.Z())
605 {
606 other.Set(0, 1, 0);
607 }
608 else
609 {
610 other.Set(0, 0, 1);
611 }
612 }
613 }
614
615 const Vector3<T> axis(_v1.Cross(other).Normalize());
616
617 qw = 0;
618 qx = axis.X();
619 qy = axis.Y();
620 qz = axis.Z();
621 }
622 else
623 {
624 // the vectors are in general position
625 const Vector3<T> axis(_v1.Cross(_v2));
626 qw = kCosTheta + k;
627 qx = axis.X();
628 qy = axis.Y();
629 qz = axis.Z();
630 this->Normalize();
631 }
632 }

900 public: static Quaternion<T> Squad(T _fT,

904 {
905 T fSlerpT = 2.0f*_fT*(1.0f-_fT);
906 Quaternion<T> kSlerpP = Slerp(_fT, _rkP, _rkQ, _shortestPath);
907 Quaternion<T> kSlerpQ = Slerp(_fT, _rkA, _rkB);
908 return Slerp(fSlerpT, kSlerpP, kSlerpQ);
909 }
910
919 public: static Quaternion<T> Slerp(T _fT,
920 const Quaternion<T> &_rkP, const Quaternion<T> &_rkQ,
921 bool _shortestPath = false)
922 {
923 T fCos = _rkP.Dot(_rkQ);
924 Quaternion<T> rkT;
925
926 // Do we need to invert rotation?
927 if (fCos < 0.0f && _shortestPath)
928 {
929 fCos = -fCos;
930 rkT = -_rkQ;
931 }
932 else
933 {
934 rkT = _rkQ;
935 }
936
937 if (std::abs(fCos) < 1 - 1e-03)
938 {
939 // Standard case (slerp)
940 T fSin = sqrt(1 - (fCos*fCos));
941 T fAngle = atan2(fSin, fCos);
942 // FIXME: should check if (std::abs(fSin) >= 1e-3)
943 T fInvSin = 1.0f / fSin;
944 T fCoeff0 = sin((1.0f - _fT) * fAngle) * fInvSin;
945 T fCoeff1 = sin(_fT * fAngle) * fInvSin;
946 return _rkP * fCoeff0 + rkT * fCoeff1;
947 }
948 else
949 {
950 // There are two situations:
951 // 1. "rkP" and "rkQ" are very close (fCos ~= +1),
952 // so we can do a linear interpolation safely.
953 // 2. "rkP" and "rkQ" are almost inverse of each
954 // other (fCos ~= -1), there
955 // are an infinite number of possibilities interpolation.
956 // but we haven't have method to fix this case, so just use
957 // linear interpolation here.
958 Quaternion<T> t = _rkP * (1.0f - _fT) + rkT * _fT;
959 // taking the complement requires renormalisation
960 t.Normalize();
961 return t;
962 }
963 }
964
973 public: Quaternion<T> Integrate(const Vector3<T> &_angularVelocity,
974 const T _deltaT) const
975 {
976 Quaternion<T> deltaQ;
977 Vector3<T> theta = _angularVelocity * _deltaT / 2;
978 T thetaMagSq = theta.SquaredLength();
979 T s;
980 if (thetaMagSq * thetaMagSq / 24.0 < MIN_D)
981 {
982 deltaQ.W() = 1.0 - thetaMagSq / 2.0;
983 s = 1.0 - thetaMagSq / 6.0;
984 }
985 else
986 {
987 double thetaMag = sqrt(thetaMagSq);
988 deltaQ.W() = cos(thetaMag);
989 s = sin(thetaMag) / thetaMag;
990 }
991 deltaQ.X() = theta.X() * s;
992 deltaQ.Y() = theta.Y() * s;
993 deltaQ.Z() = theta.Z() * s;
994 return deltaQ * (*this);
995 }
