202 public: void SetFromAxisAngle(const Vector3<T> &_axis, T _angle)
203 {
204 T c = cos(_angle);
205 T s = sin(_angle);
206 T C = 1-c;
207
208 this->data[0][0] = _axis.X()*_axis.X()*C + c;
209 this->data[0][1] = _axis.X()*_axis.Y()*C - _axis.Z()*s;
210 this->data[0][2] = _axis.X()*_axis.Z()*C + _axis.Y()*s;
211
212 this->data[1][0] = _axis.Y()*_axis.X()*C + _axis.Z()*s;
213 this->data[1][1] = _axis.Y()*_axis.Y()*C + c;
214 this->data[1][2] = _axis.Y()*_axis.Z()*C - _axis.X()*s;
215
216 this->data[2][0] = _axis.Z()*_axis.X()*C - _axis.Y()*s;
217 this->data[2][1] = _axis.Z()*_axis.Y()*C + _axis.X()*s;
218 this->data[2][2] = _axis.Z()*_axis.Z()*C + c;
219 }

227 public: void SetFrom2Axes(const Vector3<T> &_v1, const Vector3<T> &_v2)
228 {
229 const T _v1LengthSquared = _v1.SquaredLength();
230 if (_v1LengthSquared <= 0.0)
231 {
232 // zero vector - we can't handle this
233 this->Set(1, 0, 0, 0, 1, 0, 0, 0, 1);
234 return;
235 }
236
237 const T _v2LengthSquared = _v2.SquaredLength();
238 if (_v2LengthSquared <= 0.0)
239 {
240 // zero vector - we can't handle this
241 this->Set(1, 0, 0, 0, 1, 0, 0, 0, 1);
242 return;
243 }
244
245 const T dot = _v1.Dot(_v2) / sqrt(_v1LengthSquared * _v2LengthSquared);
246 if (fabs(dot - 1.0) <= 1e-6)
247 {
248 // the vectors are parallel
249 this->Set(1, 0, 0, 0, 1, 0, 0, 0, 1);
250 return;
251 }
252 else if (fabs(dot + 1.0) <= 1e-6)
253 {
254 // the vectors are opposite
255 this->Set(-1, 0, 0, 0, -1, 0, 0, 0, -1);
256 return;
257 }
258
259 const Vector3<T> cross = _v1.Cross(_v2).Normalize();
260
261 this->SetFromAxisAngle(cross, acos(dot));
262 }

476 public: T Determinant() const
477 {
478 T t0 = this->data[2][2]*this->data[1][1]
479 - this->data[2][1]*this->data[1][2];
480
481 T t1 = -(this->data[2][2]*this->data[1][0]
482 -this->data[2][0]*this->data[1][2]);
483
484 T t2 = this->data[2][1]*this->data[1][0]
485 - this->data[2][0]*this->data[1][1];
486
487 return t0 * this->data[0][0]
488 + t1 * this->data[0][1]
489 + t2 * this->data[0][2];
490 }

494 public: Matrix3<T> Inverse() const
495 {
496 T t0 = this->data[2][2]*this->data[1][1] -
497 this->data[2][1]*this->data[1][2];
498
499 T t1 = -(this->data[2][2]*this->data[1][0] -
500 this->data[2][0]*this->data[1][2]);
501
502 T t2 = this->data[2][1]*this->data[1][0] -
503 this->data[2][0]*this->data[1][1];
504
505 T invDet = 1.0 / (t0 * this->data[0][0] +
506 t1 * this->data[0][1] +
507 t2 * this->data[0][2]);
508
509 return invDet * Matrix3<T>(
510 t0,
511 - (this->data[2][2] * this->data[0][1] -
512 this->data[2][1] * this->data[0][2]),
513 + (this->data[1][2] * this->data[0][1] -
514 this->data[1][1] * this->data[0][2]),
515 t1,
516 + (this->data[2][2] * this->data[0][0] -
517 this->data[2][0] * this->data[0][2]),
518 - (this->data[1][2] * this->data[0][0] -
519 this->data[1][0] * this->data[0][2]),
520 t2,
521 - (this->data[2][1] * this->data[0][0] -
522 this->data[2][0] * this->data[0][1]),
523 + (this->data[1][1] * this->data[0][0] -
524 this->data[1][0] * this->data[0][1]));
525 }
