222 public: Vector3<T> CoordPositionAdd(const Vector3<T> &_pos) const
223 {
224 Quaternion<T> tmp(0.0, _pos.X(), _pos.Y(), _pos.Z());
225
226 // result = pose.q + pose.q * this->p * pose.q!
227 tmp = this->q * (tmp * this->q.Inverse());
228
229 return Vector3<T>(this->p.X() + tmp.X(),
230 this->p.Y() + tmp.Y(),
231 this->p.Z() + tmp.Z());
232 }
233
237 public: Vector3<T> CoordPositionAdd(const Pose3<T> &_pose) const
238 {
239 Quaternion<T> tmp(static_cast<T>(0),
240 this->p.X(), this->p.Y(), this->p.Z());
241
242 // result = _pose.q + _pose.q * this->p * _pose.q!
243 tmp = _pose.q * (tmp * _pose.q.Inverse());
244
245 return Vector3<T>(_pose.p.X() + tmp.X(),
246 _pose.p.Y() + tmp.Y(),
247 _pose.p.Z() + tmp.Z());
248 }

253 public: inline Vector3<T> CoordPositionSub(const Pose3<T> &_pose) const
254 {
255 Quaternion<T> tmp(0,
256 this->p.X() - _pose.p.X(),
257 this->p.Y() - _pose.p.Y(),
258 this->p.Z() - _pose.p.Z());
259
260 tmp = _pose.q.Inverse() * (tmp * _pose.q);
261 return Vector3<T>(tmp.X(), tmp.Y(), tmp.Z());
262 }

286 // \return The inverse pose.
287 public: Pose3<T> CoordPoseSolve(const Pose3<T> &_b) const
288 {
289 Quaternion<T> qt;
290 Pose3<T> a;
291
292 a.q = this->q.Inverse() * _b.q;
293 qt = a.q * Quaternion<T>(0, this->p.X(), this->p.Y(), this->p.Z());
294 qt = qt * a.q.Inverse();
295 a.p = _b.p - Vector3<T>(qt.X(), qt.Y(), qt.Z());
296
297 return a;
298 }

312 public: Pose3<T> RotatePositionAboutOrigin(const Quaternion<T> &_q) const
313 {
314 Pose3<T> a = *this;
315 a.p.X((1.0 - 2.0*_q.Y()*_q.Y() - 2.0*_q.Z()*_q.Z()) * this->p.X()
316 +(2.0*(_q.X()*_q.Y()+_q.W()*_q.Z())) * this->p.Y()
317 +(2.0*(_q.X()*_q.Z()-_q.W()*_q.Y())) * this->p.Z());
318 a.p.Y((2.0*(_q.X()*_q.Y()-_q.W()*_q.Z())) * this->p.X()
319 +(1.0 - 2.0*_q.X()*_q.X() - 2.0*_q.Z()*_q.Z()) * this->p.Y()
320 +(2.0*(_q.Y()*_q.Z()+_q.W()*_q.X())) * this->p.Z());
321 a.p.Z((2.0*(_q.X()*_q.Z()+_q.W()*_q.Y())) * this->p.X()
322 +(2.0*(_q.Y()*_q.Z()-_q.W()*_q.X())) * this->p.Y()
323 +(1.0 - 2.0*_q.X()*_q.X() - 2.0*_q.Y()*_q.Y()) * this->p.Z());
324 return a;
325 }
