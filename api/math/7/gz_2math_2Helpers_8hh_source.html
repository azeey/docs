369  {
370  T avg = mean<T>(_values);
371 
372  T sum = 0;
373  for (unsigned int i = 0; i < _values.size(); ++i)

538  inline int roundUpMultiple(int _num, int _multiple)
539  {
540  if (_multiple == 0)
541  return _num;
542 
543  int remainder = std::abs(_num) % _multiple;
544  if (remainder == 0)
